No need to check all cases. Use string method str.lower(). It transforms the string into lowercase and you may check word presence only once: 

cond5 = 'password' not in password.lower()

def is_acceptable_password(password: str) -> bool:

    cond1 = len(password) > 6
    cond2 = any(char.isdigit() for char in password)
    cond3 = any(char.isalpha() for char in password)
    if len(password) > 9: # cond4    
        cond2 = cond3 = True
    cond5 = 'password' not in password.lower()
        
    return all([cond1, cond2, cond3, cond5])

Solution with opposite style: checking each condition, if it is not True, immediate False return. If they all True (what you need) - return True at the end: 

def is_acceptable_password(password: str) -> bool:

    if not len(password) > 6:
        return False
    if not len(password) > 9:
        if not any(x.isdigit() for x in password):
            return False
        if not any(x.isalpha() for x in password):
            return False
    if 'password' in password.lower():
        return False

    return True
